# Heavy dependencies installed on the GPU VM by startup.sh.
# The DLVM PyTorch image (pytorch-2-7-cu128-ubuntu-2204-nvidia-570) ships
# torch + cuda + a usable Python; we layer the ASR-specific stack on top.

# Whisper
faster-whisper>=1.0.3

# HF stack — transformers 5.x has native support for Gemma 4
# (Gemma4ForConditionalGeneration). Qwen3-ASR (`qwen3_asr` model type) is
# NOT yet recognized in 5.5.0 — needs transformers main or a fork; skip
# from sweeps for now.
transformers>=5.5.0
accelerate>=0.30.0
# Gemma 4's processor uses chat templates that require jinja2 >= 3.1; the
# DLVM image ships 3.0.3 which fails with 'apply_chat_template requires
# jinja2 >= 3.1.0'.
jinja2>=3.1.0
sentencepiece>=0.2.0
protobuf>=4.25.0
safetensors>=0.4.3

# NeMo Parakeet
# NeMo 2.7 still references `np.sctypes`, removed in NumPy 2.0 — pin numpy<2.

numpy<2.0

# Datasets + audio

# torchcodec, which requires a precise FFmpeg shared-library version that the
# DLVM image doesn't ship. Older datasets (3.x) decodes via soundfile which is
# self-contained.
datasets>=3.0,<4.0
soundfile>=0.12.1
librosa>=0.10.2
